图例

1.改变图例中符号大小

% 绘图
plot(1:10,'.');

% 调整图例中符号大小
[a,b,c,d]=legend('HappyLaber');
for i=1:length(b)
    if sum(strcmp(properties(b(i)),'MarkerSize'))
       b(i).MarkerSize=20;
    end
end

% 调节尺寸
set(gcf,'Units','centimeters','Position',[8 5 15 8]);

% 导出图片
exportgraphics(gcf,'001.png','Resolution',600);

1.2 共用图例

% 布局
h=tiledlayout(1,2);
ax1=nexttile(1);
plot(1:10);
ax2=nexttile(2);
plot(1:10);

% 共用图例
lgd=legend('y=x');
lgd.Layout.Tile = 'north';

% 调节尺寸
set(gcf,'Units','centimeters','Position',[8 5 15 8]);

% 导出图片
exportgraphics(gcf,'001.png','Resolution',600);

1.4 画两个图例

% 数据
a=0:0.01:10;
y1=sin(a);
y2=cos(a);
% 绘图
p=plot(a,[y1;y2]);
% 图例
legend(p(1),'y=sinx');
ax=axes('position',get(gca,'position'),'visible','off');
legend(ax,p(2),'y=cosx','Location','northwest');
